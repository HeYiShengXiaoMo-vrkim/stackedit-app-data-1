void deleteChar(char *s, char c){
    char* source = s;
    char* destination = s;
    while(*source){
        if(*source != c){
            *destination = *source;
            destination++;
        }
        source++;
    }
    *destination = '\0';
}

#include <iostream>
using namespace std;
void deleteChar(char *s, char c);
int main()
{
  char *str, ch;
  str=new char[101];
  cin.getline(str,101);
  cin>>ch;
  deleteChar(str, ch);
  cout<<str;
  delete []str;
  return 0;
}
