
## 单向链表的讲解
```c

#include <stdio.h>
#include <stdlib.h>

// 节点结构
struct ListNode {
    int val;
    struct ListNode *next;
};

// 初始化节点
struct ListNode* createNode(int val) {
    struct ListNode* newNode = (struct ListNode*)malloc(sizeof(struct ListNode));
    if (newNode == NULL) {
        printf("内存分配失败\n");
        exit(1);
    }
    newNode->val = val;
    newNode->next = NULL;
    return newNode;
}

// 清空链表
void clear(struct ListNode **head) {
    while (*head != NULL) {
        struct ListNode* temp = *head;
        *head = (*head)->next;
        free(temp);
    }
}

// 初始化链表
void init(struct ListNode **head, int nums[], int size) {
    clear(head); // 清空链表
    if (size == 0) return;
    *head = createNode(nums[0]);
    struct ListNode* current = *head;
    for (int i = 1; i < size; i++) {
        current->next = createNode(nums[i]);
        current = current->next;
    }
}

// 遍历链表
void traverse(struct ListNode *head) {
    while (head != NULL) {
        printf("%d ", head->val);
        head = head->next;
    }
    printf("\n");
}

// 节点查找
struct ListNode* search(struct ListNode *head, int target) {
    while (head != NULL) {
        if (head->val == target) {
            return head;
        }
        head = head->next;
    }
    return NULL;
}

// 节点插入
void insert(struct ListNode **head, int index, int val) {
    if (index < 0) return;
    struct ListNode* newNode = createNode(val);
    if (index == 0) {
        newNode->next = *head;
        *head = newNode;
        return;
    }
    struct ListNode* current = *head;
    for (int i = 0; i < index - 1 && current != NULL; i++) {
        current = current->next;
    }
    if (current == NULL) return;
    newNode->next = current->next;
    current->next = newNode;
}

// 节点删除
void deleteNode(struct ListNode **head, int val) {
    if (*head == NULL) return;
    if ((*head)->val == val) {
        struct ListNode* temp = *head;
        *head = (*head)->next;
        free(temp);
        return;
    }
    struct ListNode* current = *head;
    while (current->next != NULL) {
        if (current->next->val == val) {
            struct ListNode* temp = current->next;
            current->next = current->next->next;
            free(temp);
            return;
        }
        current = current->next;
    }
}

int main() {
    struct ListNode* head = NULL;
    int nums[] = {1, 2, 3, 4, 5};
    int size = sizeof(nums) / sizeof(nums[0]);

    // 初始化链表
    init(&head, nums, size);

    // 遍历链表
    printf("遍历链表: ");
    traverse(head);

    // 查找节点
    int target = 3;
    struct ListNode* foundNode = search(head, target);
    if (foundNode != NULL) {
        printf("查找到节点 %d\n", target);
    } else {
        printf("未找到节点 %d\n", target);
    }

    // 插入节点
    int insertVal = 6;
    int insertIndex = 2;
    insert(&head, insertIndex, insertVal);
    printf("在位置 %d 插入节点 %d\n", insertIndex, insertVal);
    printf("遍历链表: ");
    traverse(head);

    // 删除节点
    int deleteVal = 4;
    deleteNode(&head, deleteVal);
    printf("删除节点 %d\n", deleteVal);
    printf("遍历链表: ");
    traverse(head);

    // 清空链表
    clear(&head);
    printf("清空链表后: ");
    traverse(head);

    return 0;
}
```
